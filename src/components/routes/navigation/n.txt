import lightIcon from "../../images/icons8-sun.svg";
import darkIcon from "../../images/icons8-code.svg";
import "../navigation/navigation.module.css";
import { Link, useLoaderData} from "react-router-dom";

import { getAbout } from "../about/about";
export async function loader() {
  const aboutPage = await getAbout();
  return { aboutPage };
}

function Navigation() {
  const { aboutPage } = useLoaderData();
  return (
    <>
      <nav className="Navigation">
        <img src={darkIcon} />
        <div className='navLinks'>
          <ul>
            {aboutPage.map((getAbout)=>
            (
              <li key={getAbout.id}>
              <Link to={`about/1`}>About</Link>
              </li>

            ) 
            )}
           
            <li>Projects</li>
            <li>Contact</li>
          </ul>
          <img src={lightIcon} />
        </div>
      </nav>
      <h3>building a portfolio</h3>
    </>
  );
}

export default Navigation;
